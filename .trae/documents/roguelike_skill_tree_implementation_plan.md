## 五、技能配置示例

### 5.1 被动技能示例 (skill_rgl_800001001.lua)
```lua
-- 连击基础: 普攻20%概率触发额外攻击
local skill_rgl_800001001 = {
    -- 技能基本信息
    id = 800001001,
    name = "连击基础",
    type = "passive",
    category = "attack",
    subcategory = "combo",
    
    -- 触发配置
    trigger = {
        timing = E_TRIGGER_TIMING.ON_ATTACK,  -- 攻击后触发
        probability = 2000,                      -- 20%概率 (万分比)
        condition = function(hero, target, skill)
            -- 只有普攻触发
            return skill.skillType == E_SKILL_TYPE_NORMAL
        end
    },
    
    -- 效果配置
    effects = {
        {
            type = "combo_attack",                -- 连击效果
            target = "self",                       -- 对自己触发
            value = 10000,                         -- 100%普攻伤害
            maxCombo = 1,                          -- 最大连击次数
        },
        {
            type = "apply_buff",                  -- 附加Buff
            buffId = 810001,                       -- 连击印记
            duration = 1,
        }
    },
    
    -- 升级配置
    upgrades = {
        [800001002] = { probability = 3500 },    -- 连击概率+: 35%
        [800001003] = { maxCombo = 2 },          -- 连击次数+1: 可连击2次
    }
}

return skill_rgl_800001001
```

### 5.2 主动技能示例 (skill_rgl_800010101.lua)
```lua
-- 火球术: 150%单体火伤
local skill_rgl_800010101 = {
    id = 800010101,
    name = "火球术",
    type = "active",
    category = "magic",
    subcategory = "fire",
    
    -- 基础配置
    baseConfig = {
        skillType = E_SKILL_TYPE_NORMAL,         -- 普攻类型
        cooldown = 0,                            -- 无冷却
        cost = 0,                                -- 无消耗
        damageRate = 15000,                      -- 150%伤害
        damageType = E_DAMAGE_TYPE.MAGIC,        -- 法术伤害
        elementType = "fire",                    -- 火元素
    },
    
    -- 目标选择
    targetSelection = {
        castTarget = E_CAST_TARGET.Enemy,        -- 敌方
        num = 1,                                 -- 1个目标
        wpType = nil,                            -- 无站位限制
        filter = nil,                            -- 无特殊过滤
    },
    
    -- 动作时间轴
    actData = {
        {
            atLeastTimeS = 1.0,
            keyFrameDatas = {
                {
                    TriggerS = 0.5,
                    datatype = "DWCommon.DamageData",
                    data = "{ damageRate = 15000, damageType = 2, elementType = 'fire' }"
                }
            }
        }
    },
    
    -- 元素加成
    elementBonus = {
        buffId = 810025,                         -- 火元素亲和Buff加成
        bonusPerStack = 500,                     -- 每层+5%伤害
    }
}

return skill_rgl_800010101
```
